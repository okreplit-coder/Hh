local function makeDraggable(frame, dragHandle)
    local isDragging = false
    local dragStart = nil
    local startPos = nil
    
    local function updatePosition(input)
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
    
    dragHandle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            isDragging = true
            dragStart = input.Position
            startPos = frame.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    isDragging = false
                end
            end)
        end
    end)
    
    dragHandle.InputChanged:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) and isDragging then
            updatePosition(input)
        end
    end)
end

local function isValidName(name)
    if not name or type(name) ~= "string" or string.len(name) < 3 then
        return false
    end
    
    local invalidPatterns = {
        "^%s*$", "^%d+$", "^[%p%s]*$",
        "Workspace", "ReplicatedStorage", "ServerStorage", "Players",
        "Lighting", "SoundService", "StarterGui", "StarterPack",
        "Teams", "TeleportService", "UserInputService", "RunService",
        "HttpService", "TweenService"
    }
    
    for _, pattern in ipairs(invalidPatterns) do
        if string.match(name, pattern) then
            return false
        end
    end
    
    return true
end

local function matchesPetName(instanceName, targetName)
    if not isValidName(instanceName) then
        return false
    end
    
    if instanceName == targetName then
        return true
    end
    
    local cleanName = string.gsub(instanceName, "%s*%([^)]*%)%s*", "")
    cleanName = string.gsub(cleanName, "%s*%d+%s*$", "")
    cleanName = string.gsub(cleanName, "^%s*(.-)%s*$", "%1")
    
    if cleanName == targetName then
        return true
    end
    
    local instanceWords = {}
    for word in string.gmatch(cleanName, "%S+") do
        table.insert(instanceWords, string.lower(word))
    end
    
    local targetWords = {}
    for word in string.gmatch(targetName, "%S+") do
        table.insert(targetWords, string.lower(word))
    end
    
    if #instanceWords >= 2 and #targetWords >= 2 then
        local matchCount = 0
        for _, targetWord in ipairs(targetWords) do
            for _, instanceWord in ipairs(instanceWords) do
                if instanceWord == targetWord then
                    matchCount = matchCount + 1
                    break
                end
            end
        end
        
        if matchCount >= math.min(2, #targetWords) then
            return true
        end
    end
    
    return false
end
